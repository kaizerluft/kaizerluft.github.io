wstring – структура данных «широких» локализованных символов, представляющих интернациональную кодировку символами Unicode


setlocale(LC_ALL, ".866"); – добавить кодировку с русскими символами
wstring Wstr = L"текст"; – объявить и инициализировать широкую строку
wcin >> Wstr; – ввести широкую строку
wcout << Wstr; – вывести широкую строку
wcout << L"текст"; – вывести русский текст
